#include <iostream>

struct User {
    int x, y;

        return x > u.x;
    }

    return a.x > b.x;
}

int main() {
    
    for (int i = 0; i < 5; i++) {
        a[i].x = i;
        a[i].y = -i;

    sort(a, a + 5);

class Solution {
public:
    void reOrderArray(vector<int> &array) {
         int i = 0, j = array.size() - 1;
         
         while (i < j) {
             while (i < j && array[i] % 2 == 1) i++;
             while (i < j && array[j] % 2 == 0) j--;
             swap(array[i], array[j]);
             i++;
             j--;
         }
         return;
    }
};

class Solution {
public:
    int NumberOf1(int n) {
        int cnt = 0;
        for (int i = 0; i< 32; i++) {
            if (n & 1) cnt++;
            n >>= 1;
        }
        return cnt;
    }
};
